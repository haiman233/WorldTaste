function onUse(event){

    var player = event.getPlayer();
    var inv = player.getInventory();
    var location = player.getLocation();
    var x1 = location.getX();
    var y1 = location.getY();
    var z1 = location.getZ();
    var itemInMainHand = inv.getItemInMainHand();
    if (itemInMainHand != null && itemInMainHand.getAmount() > 0) {
        var amount = 1;
        const radius = 4;
        const yradius = 3;
        itemInMainHand.setAmount(itemInMainHand.getAmount() - 1);
        for (let y = -radius; y <= radius; y++) {
            for (let x = -radius; x <= radius; x++) {
                for (let z = -radius; z <= radius; z++) {
                    player.getWorld().spawnParticle(org.bukkit.Particle.EXPLOSION_EMITTER, new org.bukkit.Location(player.getWorld(), x1 + x, y1 + y, z1 + z), 1);
                }
            }
        }
    }
}
